

# Renounce the dynasty
ftr_renounce_dynasty_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_realm.dds"
	}
	decision_group_type = major
	desc = ftr_renounce_dynasty_decision_desc
	selection_tooltip = ftr_renounce_dynasty_decision_tooltip
	is_shown = {
		is_landed = yes
		exists = house
		exists = house.house_head
		is_house_head = yes
		is_dynast = no
		primary_title.tier > tier_barony
		NOR = {
			has_government = holy_order_government
			has_government = theocracy_government
			AND = {
				faith = { has_doctrine = doctrine_theocracy_temporal }
				has_council_position = councillor_court_chaplain
			}
		}
		OR = {
			has_trait = ambitious
			has_trait = disinherited
			has_trait = disputed_heritage
			has_trait = wild_oat
			has_trait = denounced
		}
		NOT = {
			has_character_flag = flag_ftr_create_new_dynasty
		}
	}

	is_valid = {
		dynasty.dynast = {
			government_allows = create_cadet_branches
			custom_description = {
				text = create_cadet_branch_decision_succession_line
				NOT = {
					any_held_title = {
						place_in_line_of_succession = {
							target = root
							value <=3
						}
					}
				}
			}
		}
		dynasty = {
			dynasty_prestige_level >= 4
		}
		prestige_level >= 4
	}

	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
	}

	effect = {
		found_new_dynasty_effect = {
			PRESTIGE = major_prestige_gain
		}
		hidden_effect = {
			add_character_flag = {
				flag = flag_ftr_create_new_dynasty
			}
		}
		if = {
			limit = { has_trait = disinherited }
			remove_trait = disinherited
		}
		if = {
			limit = { has_trait = disputed_heritage }
			remove_trait = disputed_heritage
		}
		if = {
			limit = { has_trait = denounced }
			remove_trait = denounced
		}
	}

	ai_check_interval = 120
	ai_potential = {
		always = yes
	}
	ai_will_do = {
		base = -20
		modifier = {
			add = 30
			is_independent_ruler = yes
		}
		modifier = {
			add = -50
			is_independent_ruler = no
			any_liege_or_above = {
				dynasty = root.dynasty
			}
		}
		modifier = {
			factor = 0
			any_child = {
				is_alive = yes
				count < 3
			}
		}
		modifier = {
			add = 20
			OR = {
				has_trait = disinherited
				has_trait = disputed_heritage
				has_trait = wild_oat
				has_trait = denounced
			}
		}
		modifier = {
			add = -20
			OR = {
				has_trait = humble
				has_trait = content
				has_trait = loyal
				has_trait = lazy
			}
		}
		modifier = {
			add = {
				value = prestige_level
				subtract = 3
				multiply = 5
			}
		}
	}
}
